df %>% 
  clean_names() %>% 
  select(-(x10:my_home_avtar)) %>% 
  filter(!is.na(final_score_aparna),
         final_score_aparna != "-") -> df
df %>% 
  filter(weight_importance != "Total") -> df
df %>% 
  mutate(aparna_score = as.numeric(aparna_score),
         prestige_score = as.numeric(prestige_score),
         weight_importance = as.numeric(weight_importance),
         final_score_aparna = as.numeric(final_score_aparna),
         final_score_prestige = as.numeric(final_score_prestige),
         cost_factors = factor(cost_factors, cost_factors)) -> df
df %>% 
  mutate(cost_factors = reorder(cost_factors, weight_importance)) %>% 
  ggplot( aes(x=cost_factors, y=weight_importance)) +
  geom_segment( aes(x=cost_factors, xend=cost_factors, y=0, yend=weight_importance), color="skyblue", size=1) +
  geom_point( color="blue", size=4, alpha=0.6) +
  coord_flip() + 
  theme(panel.grid.minor.y = element_blank(),
        panel.grid.major.y = element_blank(),
        legend.position="none") +
  labs(x = "",
       y = "Weight Importance",
       title = "Cost Factor vs. Weight Importance",
       subtitle = "All cost factors",
       caption = "Based on data from House Buying Sheet") -> p1
p1

df %>% 
  mutate(aparna_better_than_prestige = aparna_score/prestige_score,
         cost_factors = reorder(cost_factors, aparna_better_than_prestige)) %>% 
  ggplot( aes(x=cost_factors, y=aparna_better_than_prestige)) +
  geom_segment( aes(x=cost_factors, xend=cost_factors, y=0, yend=aparna_better_than_prestige), color="skyblue", size=1) +
